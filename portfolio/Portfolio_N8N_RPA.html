④-2 에러 처리 및 안정성 확보 / Error Handling & Stability

🛡️ 에러 처리 및 안정성 확보▼

네트워크 오류 대응
// 재시도 로직
if | ${network_error} == "true"
  pause | 5000
  refresh
  pause | 3000
endif
요소 존재 확인
// 요소 존재 여부 체크
storeElementPresent | //div[@id="target-element"] | element_exists
if | "${element_exists}" == "true"
  click | //div[@id="target-element"]
endif
명시적 대기
// 페이지 로딩 대기
waitForElementPresent | //div[@id="content"] | 30000
pause | 2000

// 동적 콘텐츠 로딩 대기
waitForNotVisible | //img[@alt="로딩중"] | 10000
조건부 대기
// 여러 조건 중 하나 대기
waitForHasElements | id=loginID||link=로그아웃 | login_status
if | "${login_status}" == "id=loginID"
  // 로그인 필요
endif
데이터 정제
// 특수문자 제거
storeEval | storedVars.data.replace(/[/+~():&\n]/g, '') | clean_data

// 빈 값 체크
if | "${clean_data}" == "" || "${clean_data}" == " "
  store | "N/A" | clean_data
endif
데이터 유효성 검사
// 숫자 데이터 검증
storeEval | storedVars.count.includes("count") | is_count
if | !${is_count}
  store | 0 | count
endif

// 날짜 형식 검증
storeEval | storedVars.date.split(" - ") | date_parts
storeEval | storedVars.date_parts[0] | valid_date
// 진행 상황 로깅
echo | =========================================================================
echo | [${current_page} / ${total_pages}] - [${date}]
echo | =========================================================================

// 데이터 수집 로그
echo | (${type}, ${date}, ${key}, ${count}, ${summary}) [${cnt}/${row_cnt}]

// 에러 발생 시 로그
if | ${error_occurred} == "true"
  echo | ERROR: ${error_message}
  echo | Continuing with next item...
endif

수익 계산 공식 (프로토타입)
수익 = 총 수입 - 직접 비용
총 수입: 결제수수료 + 플랫폼수수료 + 부가서비스 + 광고수수료 + 기타수입
직접 비용: 인프라비용 + 지원금 + 마케팅비용 + 상품원가
주요 수입 항목
  • 결제수수료 수입 (거래액 기반)
  • 플랫폼 수수료 (서비스 제공)
  • 부가서비스 수입 (부가 기능)
